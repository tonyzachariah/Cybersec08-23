---
- name: Config ELK
  hosts: elk
  become: true
  tasks:
    - name: setting max memory
      ansible.posix.sysctl:
        name: vm.max_map_count
        value: '262144' 
    
    - name: docker.io
      apt:
        update_cache: yes
        name: docker.io
        state: present

    - name: Install pip3
      apt:
        name: python3-pip
        state: present

    - name: Install Docker python module
      pip:
        name: docker
        state: present

    - name: download sebp
      docker_container:
        name: sebp
        image: sebp/elk:761
        state: started
        restart_policy: always
        published_ports: 
         -5601:5601 
         -9200:9200 
         -5044:5044

    - name: Enable docker service
      systemd:
        name: docker
        enabled: yes
